StepperOutput CalculateStepper(float syringe_diameter, int diam_idx,
		float volume, int vol_idx, float flowrate, int flow_time_idx) {
	StepperOutput result;

	// 1. Convert Diameter to mm
	float diameter_mm = syringe_diameter;
	//Case: um
	if (diam_idx == 1) {
		diameter_mm = syringe_diameter / 1000.0f;
	}

	// 2. conversions for flowrate of diff units
	float volume_ml = volume;
	//case: uL/sec
	if (vol_idx == 1 && flow_time_idx == 0) {
		volume_ml = volume / 1000.0f;
		flowrate = flowrate / 1000.0f;
	}
	//Case: uL/hr
	else if (vol_idx == 1 && flow_time_idx == 1) {
		volume_ml = volume / 1000.0f;
		flowrate = flowrate / 1000.0f;
		flowrate = flowrate / 3600.0f;
	}
	//Case: mL/hr
	else if (vol_idx == 0 && flow_time_idx == 1) {
		flowrate = flowrate / 3600.0f;
	}

	// 3. Convert Flow Rate to mL/sec
	float flowrate_ml_per_s = flowrate;

	// --- Core Calculations ---
	float area = M_PI * pow((diameter_mm / 2.0f), 2);

	// displacement (mm) = (volume in mm^3) / area (1 mL = 1000 mm^3)
	float displacement_mm = (volume_ml * 1000.0f) / area;
	float lead_screw_pitch = 4.0f;

	// Ensure steps_per_rev_lead is defined globally as 3200.0f
	result.steps = (int) ((displacement_mm / lead_screw_pitch)
			* steps_per_rev_lead);

	// Calculate pulse frequency (steps per second)
	float rps = (flowrate_ml_per_s * 1000.0f) / area / lead_screw_pitch;
	result.steps_per_second = rps * steps_per_rev_lead;

	return result;
}

// Sets hardware timer frequency based on steps per second
void StartLeadMotorTimer(float steps_per_second, uint8_t direction) {
	HAL_TIM_Base_Stop_IT(&htim3);

	// Math: Period (ARR) = (Timer Clock / Steps Per Second) - 1
	// Assuming 1 MHz timer clock (PSC set to 83 for 84MHz clock)
	uint32_t arr_val = (uint32_t) (1000000.0f / steps_per_second) - 1;

	__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
	__HAL_TIM_SET_COUNTER(&htim3, 0);

	// Hardware Pin setup
	HAL_GPIO_WritePin(DIR_LEAD_PORT, DIR_LEAD_PIN, direction);
	HAL_GPIO_WritePin(EN_LEAD_PORT, EN_LEAD_PIN, GPIO_PIN_RESET); // Enable ON

	lead_dir = direction;
	lead_is_moving = 1;
	HAL_TIM_Base_Start_IT(&htim3);
}

void SeekSyringeButt(float target_sps) {
	float current_sps = 400.0f; // Start slow for high torque
	lead_mode = MOVE_SEEK_BUTT;

	// Start at initial slow speed, Direction = 1 (Down)
	StartLeadMotorTimer(current_sps, 1);

	while (lead_is_moving) {
		// Ramp up logic
		if (current_sps < target_sps) {
			current_sps += 100.0f; // Acceleration increment
			uint32_t arr_val = (uint32_t) (1000000.0f / current_sps) - 1;
			__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
		}

		HAL_Delay(5); // Ramp smoothing delay

		// Interrupt logic handles lead_is_moving = 0 upon sensor hit
		if (!lead_is_moving)
			break;
	}
}

void DriveStepper(int steps, float steps_per_second, uint8_t direction) {
	lead_steps_remaining = steps;
	lead_mode = MOVE_EXACT_STEPS;
	StartLeadMotorTimer(steps_per_second, direction);
	while (lead_is_moving)
		; // Blocking wait for finish or limit
}

void DriveStepperRamped(int total_steps, float target_sps, uint8_t direction) {
	float current_sps = 20.0f; // Start slow for higher torque

	// Safety check: prevent division by zero or negative targets
	if (target_sps < current_sps)
		target_sps = current_sps;

	lead_mode = MOVE_EXACT_STEPS;
	lead_steps_remaining = total_steps;
	lead_is_moving = 1;

	// Start the timer at the initial slow speed
	StartLeadMotorTimer(current_sps, direction);

	while (lead_is_moving) {
		// --- 1. RAMPING LOGIC ---
		// Increase speed until we hit target_sps
		if (current_sps < target_sps) {
			current_sps += 15.0f;

			// Limit current_sps so it doesn't overshoot the target
			if (current_sps > target_sps)
				current_sps = target_sps;

			// Update the Timer Auto-Reload Register (ARR)
			// Assumes 1MHz Timer clock (1000000 / sps)
			uint32_t arr_val = (uint32_t) (1000000.0f / current_sps) - 1;
			__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
		}

		// The ISR handles lead_steps_remaining; we exit when the move is done
		if (lead_steps_remaining <= 0) {
			lead_is_moving = 0;
			break;
		}
	}

	// Final safety: ensure timer is stopped
	HAL_TIM_Base_Stop_IT(&htim3);
}

//Start timer for tray
void StartTrayMotorTimer(float rps) {
	HAL_TIM_Base_Stop_IT(&htim3);

	// 1/4 Step Math: 3200 steps per rev
	float steps_per_second = rps * steps_per_rev_lead;
	uint32_t arr_val = (uint32_t) (1000000.0f / steps_per_second) - 1;

	__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);

	HAL_GPIO_WritePin(EN_TRAY_PORT, EN_TRAY_PIN, GPIO_PIN_RESET); // Enable Tray
	tray_is_moving = 1;
	HAL_TIM_Base_Start_IT(&htim3);
}

/**
 * @brief Rotates the Tray 45 degrees (100 steps @ 1/4 step)
 * with Acceleration and Deceleration.
 */
void RotateTray45_Timer(float min_rps, float max_rps) {
	int total_steps = 100;     // 45 degrees at 1/4 step
	int accel_steps = 30;      // Steps to reach max speed
	int decel_start = 70; // Step count at which to start slowing down (100 - 30)
	float current_rps = min_rps;

	tray_steps_remaining = total_steps;
	tray_homing = 0;
	tray_is_moving = 1;

	// Start Timer at minimum speed
	StartTrayMotorTimer(current_rps);

	while (tray_is_moving) {
		int steps_taken = total_steps - tray_steps_remaining;

		// --- ACCELERATION PHASE ---
		if (steps_taken < accel_steps && current_rps < max_rps) {
			current_rps += 0.005f; // Small speed increment
		}
		// --- DECELERATION PHASE ---
		else if (steps_taken > decel_start && current_rps > min_rps) {
			current_rps -= 0.005f; // Small speed decrement
		}
		// --- CRUISE PHASE ---
		// (Just maintains max_rps)

		// Update Timer Frequency (ARR)
		// sps = rps * 800 (for 1/4 stepping)
		float sps = current_rps * 800.0f;
		uint32_t arr_val = (uint32_t) (1000000.0f / sps) - 1;
		__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);

		HAL_Delay(10); // Ramp smoothing delay

		if (tray_steps_remaining <= 0)
			break;
	}

	// Wait for the interrupt to finish the very last step
	while (tray_is_moving)
		;

	// Optional: Keep enabled for holding torque, or disable to save heat
	// HAL_GPIO_WritePin(EN_TRAY_PORT, EN_TRAY_PIN, GPIO_PIN_SET);
}

void RotateTray45_Trapezoidal(float min_rps, float max_rps) {
	int total_steps = 400;  // 3200 / 8
	int accel_steps = 100;
	int decel_start = 300;
	float current_rps = min_rps;

	tray_steps_remaining = total_steps;
	tray_homing = 0;
	StartTrayMotorTimer(current_rps);

	while (tray_is_moving) {
		int steps_taken = total_steps - tray_steps_remaining;
		if (steps_taken < accel_steps && current_rps < max_rps)
			current_rps += 0.002f;
		else if (steps_taken > decel_start && current_rps > min_rps)
			current_rps -= 0.002f;

		uint32_t arr_val = (uint32_t) (1000000.0f
				/ (current_rps * steps_per_rev_lead)) - 1;
		__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
		HAL_Delay(5);
		if (tray_steps_remaining <= 0)
			break;
	}
}

void HomeTray_Ramped(float target_rps) {
	float current_rps = 0.02f;
	tray_steps_remaining = 8000;   // Safety timeout
	tray_homing = 1;               // Enable sensor check in interrupt
	tray_clear = 0;
	tray_is_moving = 1;

	// PHASE 1: Fast Search (with ramp up)
	StartTrayMotorTimer(current_rps);
	while (tray_is_moving) {
		if (current_rps < target_rps) {
			current_rps += 0.0025f;
			uint32_t arr_val = (uint32_t) (1000000.0f
					/ (current_rps * steps_per_rev_lead)) - 1;
			__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
		}
		HAL_Delay(5);
	}

	// PHASE 2: Smooth Deceleration (The 200-step "Park")
	// The interrupt has stopped the motor. We now move 200 more steps manually.
	int decel_steps = 400;
	float start_decel_rps = current_rps;
	float min_rps = 0.04f;

	tray_steps_remaining = decel_steps;
	tray_homing = 0; // Disable sensor check so we don't trigger again

	StartTrayMotorTimer(start_decel_rps);

	while (tray_is_moving) {
		// Ramp down based on steps remaining
		if (current_rps > min_rps) {
			// Linear reduction: reduce speed as tray_steps_remaining goes to 0
			current_rps -= (start_decel_rps - min_rps) / decel_steps;
			if (current_rps < min_rps)
				current_rps = min_rps;

			uint32_t arr_val = (uint32_t) (1000000.0f
					/ (current_rps * steps_per_rev_lead)) - 1;
			__HAL_TIM_SET_AUTORELOAD(&htim3, arr_val);
		}
		HAL_Delay(2); // Faster update for smoother decel
		if (tray_steps_remaining <= 0)
			break;
	}
}

// Returns 1 if we should ABORT the synthesis, 0 if we should continue
int Check_Pause_During_Move(void) {
	uint16_t tx, ty;

	// Check if Touch IRQ is triggered (Low)
	if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1) == GPIO_PIN_RESET) {
		Touch_GetXY(&tx, &ty);

		// Is the touch inside the PAUSE button area?
		if (tx >= 110 && tx <= 210 && ty >= 200 && ty <= 230) {

			// --- 1. STOP HARDWARE ---
			HAL_TIM_Base_Stop_IT(&htim3); // Stop steps immediately
			HAL_GPIO_WritePin(GPIOC, GPIO_PIN_5, GPIO_PIN_SET); // Disable Lead Driver (Save heat)

			// --- 2. DRAW UI OVERLAY ---
			TFT_FillRect_Fast(50, 75, 220, 90, ILI9341_COLOR_BLACK);
			TFT_DrawRect(50, 75, 220, 90, ILI9341_COLOR_HALO);
			TFT_DrawString(85, 90, "SYSTEM PAUSED", ILI9341_COLOR_WHITE, 1);

			TFT_DrawRect(70, 125, 80, 30, 0x07E0); // Resume (Green)
			TFT_DrawString(85, 135, "RESUME", 0x07E0, 1);
			TFT_DrawRect(170, 125, 80, 30, 0xF800); // Exit (Red)
			TFT_DrawString(195, 135, "EXIT", 0xF800, 1);

			// --- 3. WAIT FOR INPUT ---
			while (1) {
				if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1) == GPIO_PIN_RESET) {
					Touch_GetXY(&tx, &ty);

					// RESUME Pressed
					if (tx >= 70 && tx <= 150 && ty >= 125 && ty <= 155) {
						Draw_Running_Screen_Static(); // Clear the overlay
						HAL_GPIO_WritePin(GPIOC, GPIO_PIN_5, GPIO_PIN_RESET); // Re-enable Lead
						HAL_TIM_Base_Start_IT(&htim3); // Restart steps
						return 0;
					}

					// EXIT Pressed
					if (tx >= 170 && tx <= 250 && ty >= 125 && ty <= 155) {
						currentState = MENU_MAIN;
						return 1; // Signal to kill the loop
					}
				}
			}
		}
	}
	return 0;
}

void Save_Recipe_To_Flash(int slot, Recipe *r) {
	uint32_t address = 0x080E0000 + (slot * sizeof(Recipe)); // Example Flash Address

	HAL_FLASH_Unlock();
	// 1. Erase the sector first (Crucial!)
	FLASH_Erase_Sector(FLASH_SECTOR_1, VOLTAGE_RANGE_3);

	// 2. Write the struct byte-by-byte or word-by-word
	uint32_t *dataPtr = (uint32_t*) r;
	for (int i = 0; i < (sizeof(Recipe) / 4); i++) {
		HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, address + (i * 4),
				dataPtr[i]);
	}
	HAL_FLASH_Lock();
}

/* FLASH DEFINITIONS FOR F401RE */
#define FLASH_STORAGE_ADDRESS 0x08040000 // Start of Sector 6
#define FLASH_SECTOR_NUM      FLASH_SECTOR_6

void SaveRecipesToFlash(void) {
	HAL_FLASH_Unlock();

	// 1. Erase Sector 6
	FLASH_EraseInitTypeDef EraseInitStruct;
	uint32_t SectorError;
	EraseInitStruct.TypeErase = FLASH_TYPEERASE_SECTORS;
	EraseInitStruct.VoltageRange = FLASH_VOLTAGE_RANGE_3; // 2.7V to 3.6V
	EraseInitStruct.Sector = FLASH_SECTOR_NUM;
	EraseInitStruct.NbSectors = 1;

	if (HAL_FLASHEx_Erase(&EraseInitStruct, &SectorError) != HAL_OK) {
		HAL_FLASH_Lock();
		return; // Handle error: maybe show a "Save Failed" message on LCD
	}

	// 2. Write the recipeDatabase array
	uint32_t *dataPtr = (uint32_t*) &recipeDatabase;
	uint32_t address = FLASH_STORAGE_ADDRESS;
	int wordsToWrite = sizeof(recipeDatabase) / 4;

	for (int i = 0; i < wordsToWrite; i++) {
		if (HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, address, dataPtr[i])
				== HAL_OK) {
			address += 4;
		}
	}

	// 3. Save the saved_recipe_count immediately after the array
	HAL_FLASH_Program(FLASH_TYPEPROGRAM_WORD, address,
			(uint32_t) saved_recipe_count);

	HAL_FLASH_Lock();
}

void LoadRecipesFromFlash(void) {
	uint32_t address = FLASH_STORAGE_ADDRESS;

	// Check if flash is empty (fresh chip/erased flash is all 1s)
	if (*(__IO uint32_t*) address == 0xFFFFFFFF) {
		saved_recipe_count = 0;
		return;
	}

	// Copy flash data into RAM
	memcpy(&recipeDatabase, (void*) address, sizeof(recipeDatabase));

	// Load the count from the address immediately following the database
	saved_recipe_count = (int) (*(__IO uint32_t*) (address
			+ sizeof(recipeDatabase)));

	// Safety: Cap the count to prevent array index issues
	if (saved_recipe_count > 5)
		saved_recipe_count = 5;
}

// Support float to string 4 decimal places
void ftoa(float n, char *res, int afterpoint) {
	int ipart = (int) n;
	float fpart = n - (float) ipart;

	// Handle negative numbers if necessary
	if (n < 0) {
		*res++ = '-';
		ipart = -ipart;
		fpart = -fpart;
	}

	// Convert integer part
	itoa(ipart, res, 10);
	int len = strlen(res);

	if (afterpoint > 0) {
		res[len] = '.';

		// Multiply by 10^afterpoint (e.g., 10000 for 4 places)
		float multiplier = pow(10, afterpoint);
		int fpart_int = (int) (fpart * multiplier + 0.5f); // +0.5 for rounding

		// Convert fractional part and pad with leading zeros
		char f_str[10];
		itoa(fpart_int, f_str, 10);

		int f_len = strlen(f_str);
		int i;
		// Add leading zeros if needed (e.g., .0005)
		for (i = 0; i < (afterpoint - f_len); i++) {
			res[len + 1 + i] = '0';
		}
		strcpy(res + len + 1 + i, f_str);
	}
}

// Helper to convert int to string (for Flow Rate/Time)
void itoa_simple(int n, char *res) {
	itoa(n, res, 10);
}

void TFT_FillRect_Fast(uint16_t x, uint16_t y, uint16_t w, uint16_t h,
		uint16_t color) {
	if (x >= 320 || y >= 240)
		return;
	if ((x + w) > 320)
		w = 320 - x;
	if ((y + h) > 240)
		h = 240 - y;

	TFT_SetAddressWindow(x, y, x + w - 1, y + h - 1); // Set the block area once [cite: 70-72]

	uint8_t color_high = color >> 8;
	uint8_t color_low = color & 0xFF;

	HAL_GPIO_WritePin(TFT_DC_GPIO_Port, TFT_DC_Pin, GPIO_PIN_SET);
	HAL_GPIO_WritePin(TFT_CS_GPIO_Port, TFT_CS_Pin, GPIO_PIN_RESET);

	// Write all pixels in one continuous stream
	for (uint32_t i = 0; i < (uint32_t) w * h; i++) {
		HAL_SPI_Transmit(&hspi1, &color_high, 1, 10);
		HAL_SPI_Transmit(&hspi1, &color_low, 1, 10);
	}
	HAL_GPIO_WritePin(TFT_CS_GPIO_Port, TFT_CS_Pin, GPIO_PIN_SET);
}

uint16_t Touch_ReadRaw(uint8_t cmd) {
	uint8_t tx_data[3] = { cmd, 0x00, 0x00 };
	uint8_t rx_data[3];

	HAL_GPIO_WritePin(GPIOB, TOUCH_CS_Pin, GPIO_PIN_RESET);
	HAL_SPI_TransmitReceive(&hspi1, tx_data, rx_data, 3, 10);
	HAL_GPIO_WritePin(GPIOB, TOUCH_CS_Pin, GPIO_PIN_SET);

	return ((rx_data[1] << 8) | rx_data[2]) >> 3;
}

// Returns 1 if the synthesis should be aborted entirely, 0 to continue
int Handle_Pause_Condition(void) {
	uint16_t tx, ty;

	// 1. Check if the screen is touched
	if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1) == GPIO_PIN_RESET) {
		Touch_GetXY(&tx, &ty);

		// 2. Check if touch is within PAUSE button (X: 110-210, Y: 200-230)
		if (tx >= 110 && tx <= 210 && ty >= 200 && ty <= 230) {
			currentState = MENU_PAUSED;

			// Draw Pause Overlay
			TFT_FillRect_Fast(60, 80, 200, 100, ILI9341_COLOR_BLACK);
			TFT_DrawRect(60, 80, 200, 100, ILI9341_COLOR_HALO);
			TFT_DrawString(110, 100, "SYSTEM PAUSED", ILI9341_COLOR_WHITE, 1);

			// Draw Resume and Stop buttons inside the overlay
			TFT_DrawRect(80, 130, 70, 30, 0x07E0); // Green Resume
			TFT_DrawString(90, 140, "RESUME", 0x07E0, 1);

			TFT_DrawRect(170, 130, 70, 30, 0xF800); // Red Stop
			TFT_DrawString(190, 140, "STOP", 0xF800, 1);

			// 3. Hang here until user makes a choice
			while (currentState == MENU_PAUSED) {
				if (HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_1) == GPIO_PIN_RESET) {
					Touch_GetXY(&tx, &ty);

					// Resume pressed
					if (tx >= 80 && tx <= 150 && ty >= 130 && ty <= 160) {
						currentState = MENU_RUNNING;
						// Redraw the original Running Screen to clean up the overlay
						Draw_Running_Screen_Static();
						return 0;
					}

					// Stop pressed
					if (tx >= 170 && tx <= 240 && ty >= 130 && ty <= 160) {
						currentState = MENU_MAIN;
						return 1; // Signal to abort the loop
					}
				}
			}
		}
	}
	return 0; // No pause requested
}

void Touch_GetXY(uint16_t *x, uint16_t *y) {
	uint16_t rawX = Touch_ReadRaw(0xD0);
	uint16_t rawY = Touch_ReadRaw(0x90);

	// Physical sensor limits from your debug data
	int32_t RAW_MIN_X = 236;
	int32_t RAW_MAX_X = 3830;
	int32_t RAW_MIN_Y = 376;
	int32_t RAW_MAX_Y = 3934;

	// Based on your data:
	// Left side (0) is RawY ~3934, Right side (319) is RawY ~376
	int32_t calX = 319
			- (((int32_t) rawY - RAW_MIN_Y) * 319 / (RAW_MAX_Y - RAW_MIN_Y));

	// Top side (0) is RawX ~3830, Bottom side (239) is RawX ~236
	int32_t calY = 239
			- (((int32_t) rawX - RAW_MIN_X) * 239 / (RAW_MAX_X - RAW_MIN_X));

	// Hard Clamping to prevent logic errors
	if (calX < 0)
		calX = 0;
	if (calX > 319)
		calX = 319;
	if (calY < 0)
		calY = 0;
	if (calY > 239)
		calY = 239;

	*x = (uint16_t) calX;
	*y = (uint16_t) calY;
}

// Character and Num definations
void TFT_DrawChar(uint16_t x, uint16_t y, char c, uint16_t color, int scale) {
	const uint8_t *bitmap;

	// Mapping logic
	if (c >= 'A' && c <= 'Z') {
		bitmap = font_chars[c - 'A'];
	} else if (c >= 'a' && c <= 'z') {
		bitmap = font_chars_low[c - 'a'];
	} else if (c >= '0' && c <= '9') {
		bitmap = font_nums[c - '0'];
	} else if (c == '.') {
		bitmap = font_nums[10]; // Using your array index instead of static
	} else if (c == '-') {
		bitmap = font_nums[11]; // Hyphen/Minus
	} else if (c == '<' || c == '>') {
		bitmap = font_nums[12]; // Backspace Arrow
	} else if (c == '?') {
		bitmap = font_nums[13]; // Question
	} else if (c == ' ') {
		return;
	} else {
		return;
	}

	// Drawing logic (Pixel-doubling / Scaling)
	for (int i = 0; i < 5; i++) {
		for (int j = 0; j < 8; j++) {
			if (bitmap[i] & (1 << j)) {
				TFT_FillRect(x + (i * scale), y + (j * scale), scale, scale,
						color);
			}
		}
	}
}

//Test Simulation
void SimulateFullSynthesis(void) {
	currentState = MENU_RUNNING;

	// 1. Initial UI Update
	TFT_FillRect_Fast(0, 0, 320, 240, 0x0841);
	TFT_DrawString(40, 100, "INITIALIZING SIMULATION", ILI9341_COLOR_WHITE, 2);
	HAL_Delay(1000);

	Draw_Running_Screen_Static();

	// Define the Card Background color for clean erasures
	uint16_t card_bg = 0x7BEF;

	for (int i = 0; i < 8; i++) {
		if (volume_mls[i] <= 0.01f)
			continue;

		// Update UI Values
		// Inside the for (int i = 0; i < 8; i++) loop
		Update_Running_Screen_Values(i + 1,         // Syringe Label (#1, #2...)
		volume_mls[i],                  // Current Volume
				flow_rates[i],                  // Current Flow
				i + 1,                          // Progress (1 through 8)
				vol_unit_indices[i],            // Volume unit index
				flow_time_indices[i]            // Flow time index
				);

		// --- PHASE 1: SEEKING (Responsive) ---
		TFT_FillRect_Fast(90, 145, 160, 10, card_bg); // Use card_bg, not Gray
		TFT_DrawString(90, 145, "SIMULATING: SEEKING...", 0xF800, 1);

		// Instead of HAL_Delay(800), check pause every 100ms
		for (int j = 0; j < 8; j++) {
			if (Handle_Pause_Condition())
				return;
			HAL_Delay(100);
		}

		// --- PHASE 2: DISPENSING (Responsive) ---
		TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
		TFT_DrawString(90, 145, "SIMULATING: DISPENSING...", 0xF800, 1);

		// Split 2000ms into 20 checks for a highly responsive pause button
		for (int j = 0; j < 20; j++) {
			if (Handle_Pause_Condition())
				return;
			HAL_Delay(100);
		}

		// --- PHASE 3: RETRACTING (Responsive) ---
		TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
		TFT_DrawString(90, 145, "SIMULATING: RETRACTING...", 0xF800, 1);

		for (int j = 0; j < 8; j++) {
			if (Handle_Pause_Condition())
				return;
			HAL_Delay(100);
		}

		// --- PHASE 4: WAIT FOR BREAK ---
		if (time_breaks[i] > 0) {
			TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
			char break_msg[30];
			sprintf(break_msg, "WAITING: %ds", (int) time_breaks[i]);
			TFT_DrawString(90, 145, break_msg, ILI9341_COLOR_HALO, 1);

			// Check pause once per second of the break
			for (int j = 0; j < (int) time_breaks[i]; j++) {
				if (Handle_Pause_Condition())
					return;
				HAL_Delay(1000);
			}
		}

		// --- PHASE 5: ROTATE ---
		if (i < 7) {
			TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
			TFT_DrawString(90, 145, "SIMULATING: ROTATING TRAY", 0xF800, 1);
			for (int j = 0; j < 10; j++) {
				if (Handle_Pause_Condition())
					return;
				HAL_Delay(100);
			}
		}
	}

	// --- TRANSITION TO FINISHED STATE ---
	currentState = MENU_FINISHED;
	Draw_Finished_Screen();
}

// Actual Hardware
void RunFullSynthesis(void) {
	uint16_t card_bg = 0x7BEF;
	const int MAX_SAFE_STEPS = 152000; //19mm

	// --- NEW: PRE-VALIDATION LOOP ---
	for (int i = 0; i < 8; i++) {
		// Only check slots that aren't empty
		if (volume_mls[i] > 0.01f) {
			StepperOutput check = CalculateStepper(syringe_diams[i],
					diam_unit_indices[i], volume_mls[i], vol_unit_indices[i],
					flow_rates[i], flow_time_indices[i]);

			if (check.steps > MAX_SAFE_STEPS) {
				// Display Error Screen
				TFT_FillScreen(COLOR_RED);
				TFT_DrawString(40, 60, "INVALID VALUE DETECTED",
				ILI9341_COLOR_WHITE, 2);

				char err_msg[40];
				sprintf(err_msg, "SYRINGE %d: %d STEPS", i + 1, check.steps);
				TFT_DrawString(40, 100, err_msg, ILI9341_COLOR_BLACK, 1);
				TFT_DrawString(40, 130, "EXCEEDS LIMIT!", ILI9341_COLOR_BLACK,
						1);

				HAL_Delay(3000); // Let user read the error

				// Send user back to the edit screen for that specific syringe
				current_fluid_idx = i;
				currentState = MENU_NEW_SYNTHESIS;
				Draw_New_Synthesis_Menu();
				return; // Kill the synthesis before it starts
			}
		}
	}

	//Safety check passed
	currentState = MENU_RUNNING;

	// 1. Initial UI & Homing
	TFT_FillRect_Fast(0, 0, 320, 240, 0x0841);
	TFT_DrawString(45, 110, "HOMING SYSTEM", ILI9341_COLOR_WHITE, 3);

	HAL_GPIO_WritePin(EN_TRAY_PORT, EN_TRAY_PIN, GPIO_PIN_RESET); // Enable Tray
	HomeTray_Ramped(0.15f);

	// Draw the static layout once
	Draw_Running_Screen_Static();

	for (int i = 0; i < 8; i++) {
		// Skip empty slots
		if (volume_mls[i] <= 0.01f) {
			RotateTray45_Trapezoidal(min_tray_rps, max_tray_rps);
			continue;
		}

		// --- LIVE UI UPDATE ---
		Update_Running_Screen_Values(i + 1,         // Syringe Label (#1, #2...)
		volume_mls[i],                  // Current Volume
				flow_rates[i],                  // Current Flow
				i + 1,                          // Progress (1 through 8)
				vol_unit_indices[i],            // Volume unit index
				flow_time_indices[i]            // Flow time index
				);

		// --- PHASE 1: CALCULATING ---
		// Using the global index arrays you've already defined in your UI
		StepperOutput pump = CalculateStepper(syringe_diams[i], // float diameter
				diam_unit_indices[i],        // int diam_idx
				volume_mls[i],               // float volume
				vol_unit_indices[i],         // int vol_idx
				flow_rates[i],               // float flowrate
				flow_time_indices[i]         // int flow_time_idx
				);

		// --- PHASE 2: SEEKING ---
		TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
		TFT_DrawString(90, 145, "SEEKING SYRINGE...", 0xF800, 1);

		//Move down
		if (pump.steps_per_second <= movedown_speed) {
			SeekSyringeButt(movedown_speed); //hardware
		} else {
			SeekSyringeButt(pump.steps_per_second); //hardware
		}

		TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
		TFT_DrawString(90, 145, "DISPENSING...", 0xF800, 1);
		DriveStepper(pump.steps, pump.steps_per_second, 1);

		// --- PHASE 3: RETRACTING ---
		TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
		TFT_DrawString(90, 145, "RETRACTING...", 0xF800, 1);
		RetractToTop(moveup_speed);

		// --- PHASE 4: RESPONSIVE BREAK ---
		if (time_breaks[i] > 0) {
			TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
			//Logic  for different units in timebreak
			//case: min
			if (time_unit_indices[i] == 1) {
				new_time = time_breaks[i] * 60;
			}
			//case: hr
			else if (time_unit_indices[i] == 2) {
				new_time = time_breaks[i] * 3600;
			}
			for (int seconds = 0; seconds < (int) new_time; seconds++) {
				char break_msg[20];
				sprintf(break_msg, "WAITING: %ds", (int) new_time - seconds);
				TFT_DrawString(90, 145, break_msg, ILI9341_COLOR_HALO, 1);

				HAL_Delay(1000);

				TFT_FillRect_Fast(90, 145, 160, 10, card_bg); // Clear for next second
			}
		}

		// --- PHASE 5: ROTATE ---
		if (i < 7) {
			TFT_FillRect_Fast(90, 145, 160, 10, card_bg);
			TFT_DrawString(90, 145, "ROTATING TRAY...", 0xF800, 1);
			RotateTray45_Trapezoidal(min_tray_rps, max_tray_rps);
			HAL_Delay(500); // Short settle delay

		}
	}

	// --- SUCCESS COMPLETION ---
	currentState = MENU_FINISHED;
	Draw_Finished_Screen();
	HAL_GPIO_WritePin(EN_TRAY_PORT, EN_TRAY_PIN, GPIO_PIN_SET); // Disable Motors
	return;
}

			else if (currentState == MENU_NEW_SYNTHESIS) {
				if (ty >= 25 && ty <= 55) { // Tabs
					for (int i = 0; i < 8; i++) {
						int bx_start = 95 + (i * 28);
						int bx_end = bx_start + 24;
						if (tx >= bx_start && tx <= bx_end) {
							if (current_fluid_idx != i) {
								current_fluid_idx = i;
								Update_Fluid_Selection(i);
								HAL_Delay(250);
							}
						}
					}
				} else if (tx >= 90 && tx <= 195) { // Value Boxes
					if (ty >= 65 && ty <= 92)
						currentEdit = EDIT_DIAM;
					else if (ty >= 105 && ty <= 132)
						currentEdit = EDIT_VOL;
					else if (ty >= 145 && ty <= 172)
						currentEdit = EDIT_FLOW;
					else if (ty >= 185 && ty <= 212)
						currentEdit = EDIT_TIME;

					if (currentEdit != EDIT_NONE) {
						currentState = MENU_NUMPAD;
						numpad_buffer[0] = '\0';
						Draw_Numpad();
						HAL_Delay(300);
					}
				} else if (tx >= 210 && tx <= 270) { // Units
					char flow_buf[12];
					if (ty >= 65 && ty <= 92) {
						diam_unit_indices[current_fluid_idx] =
								(diam_unit_indices[current_fluid_idx] + 1) % 2;
						Draw_Unit_Button(210, 68,
								diam_units[diam_unit_indices[current_fluid_idx]]);
					} else if (ty >= 105 && ty <= 132) {
						vol_unit_indices[current_fluid_idx] =
								(vol_unit_indices[current_fluid_idx] + 1) % 2;
						Draw_Unit_Button(210, 108,
								vol_units[vol_unit_indices[current_fluid_idx]]);
						sprintf(flow_buf, "%s/%c",
								vol_units[vol_unit_indices[current_fluid_idx]],
								(flow_time_indices[current_fluid_idx] == 0 ?
										's' : 'h'));
						Draw_Unit_Button(210, 148, flow_buf);
					} else if (ty >= 145 && ty <= 172) {
						flow_time_indices[current_fluid_idx] =
								(flow_time_indices[current_fluid_idx] + 1) % 2;
						sprintf(flow_buf, "%s/%c",
								vol_units[vol_unit_indices[current_fluid_idx]],
								(flow_time_indices[current_fluid_idx] == 0 ?
										's' : 'h'));
						Draw_Unit_Button(210, 148, flow_buf);
					} else if (ty >= 185 && ty <= 212) {
						time_unit_indices[current_fluid_idx] =
								(time_unit_indices[current_fluid_idx] + 1) % 3;
						Draw_Unit_Button(210, 188,
								time_units[time_unit_indices[current_fluid_idx]]);
					}
					HAL_Delay(200);
				} else if (ty >= 210) { //Back button
					if (tx >= 5 && tx <= 100) {
						currentState = MENU_MAIN;
						Draw_Main_Menu();
						HAL_Delay(300);
					} else if (tx >= 220 && tx <= 315) { //Run button
						currentState = MENU_CONFIRM;
						Draw_Confirm_Popup();
						HAL_Delay(300);
					}
				}
			} else if (currentState == MENU_CONFIRM) { //Confirmation of action menu
				// YES Button
				if (tx > 60 && tx < 140 && ty > 120 && ty < 160) {
					if (sim == 0)
						RunFullSynthesis();
					else
						SimulateFullSynthesis();
					HAL_Delay(300);
				}

				// NO Button
				else if (tx > 180 && tx < 260 && ty > 120 && ty < 160) {
					currentState = MENU_NEW_SYNTHESIS;
					Draw_New_Synthesis_Menu();
					HAL_Delay(300);
				}
			}

			// --- 3. NUMPAD MENU ---
			else if (currentState == MENU_NUMPAD) {
				if (ty >= 50 && ty < 210 && tx < 315) {
					int col = tx / 105;
					int row = (ty - 50) / 40;
					int idx = (row * 3) + col;
					int localX = tx % 105;
					int localY = (ty - 50) % 40;

					if (idx >= 0 && idx < 12 && localX < 95 && localY < 35) {
						char *keys[] = { "1", "2", "3", "4", "5", "6", "7", "8",
								"9", ".", "0", "<-" };
						if (strcmp(keys[idx], "<-") == 0) {
							int len = strlen(numpad_buffer);
							if (len > 0)
								numpad_buffer[len - 1] = '\0';
						} else if (strlen(numpad_buffer) < 9) {
							strcat(numpad_buffer, keys[idx]);
						}
						TFT_FillRect_Fast(11, 6, 298, 33, ILI9341_COLOR_BLACK);
						TFT_DrawString(20, 15, numpad_buffer,
						ILI9341_COLOR_WHITE, 2);
						HAL_Delay(200);
					}
				} else if (ty >= 210 && tx > 10 && tx < 310) {
					float val = atof(numpad_buffer);
					if (currentEdit == EDIT_DIAM)
						syringe_diams[current_fluid_idx] = val;
					else if (currentEdit == EDIT_VOL)
						volume_mls[current_fluid_idx] = val;
					else if (currentEdit == EDIT_FLOW)
						flow_rates[current_fluid_idx] = val;
					else if (currentEdit == EDIT_TIME)
						time_breaks[current_fluid_idx] = val;

					currentState = MENU_NEW_SYNTHESIS;
					Draw_New_Synthesis_Menu();
					HAL_Delay(300);
				}
			}

			// --- 4. FINISHED MENU ---
			else if (currentState == MENU_FINISHED) {
				// SAVE button clicked
				if (ty >= 190 && ty <= 225 && tx >= 30 && tx <= 150) {
					keyboard_buffer[0] = '\0'; // Clear the name buffer
					currentState = MENU_KEYBOARD;
					Draw_Keyboard();
					HAL_Delay(300);
				}
				// EXIT button clicked
				else if (ty >= 190 && ty <= 225 && tx >= 170 && tx <= 290) {
					currentState = MENU_MAIN;
					Draw_Main_Menu();
					HAL_Delay(300);
				}
			}

			// --- STATE: KEYBOARD (Naming the Recipe) ---
			else if (currentState == MENU_KEYBOARD) {
				// 1. Calculate which key was pressed
				int col = (tx - 5) / 31;
				int row = (ty - 85) / 45;
				int idx = (row * 10) + col;

				if (idx >= 0 && idx < 30) {
					// --- Special Case: BACKSPACE ---
					if (strcmp(keys[idx], "<") == 0) {
						int len = strlen(keyboard_buffer);
						if (len > 0)
							keyboard_buffer[len - 1] = '\0';
					}
					// --- Special Case: SAVE/OK ---
					else if (strcmp(keys[idx], "OK") == 0) {
						if (strlen(keyboard_buffer) > 0
								&& saved_recipe_count < 5) {

							// --- THE DATA SAVE (Must happen before switching menus) ---
							strncpy(recipeDatabase[saved_recipe_count].name,
									keyboard_buffer, 15);
							recipeDatabase[saved_recipe_count].name[15] = '\0';

							for (int i = 0; i < 8; i++) {
								recipeDatabase[saved_recipe_count].syringe_diams[i] =
										syringe_diams[i];
								recipeDatabase[saved_recipe_count].volume_mls[i] =
										volume_mls[i];
								recipeDatabase[saved_recipe_count].flow_rates[i] =
										flow_rates[i];
								recipeDatabase[saved_recipe_count].time_breaks[i] =
										time_breaks[i];
								recipeDatabase[saved_recipe_count].diam_unit_indices[i] =
										diam_unit_indices[i];
								recipeDatabase[saved_recipe_count].vol_unit_indices[i] =
										vol_unit_indices[i];
								recipeDatabase[saved_recipe_count].flow_time_indices[i] =
										flow_time_indices[i];
								recipeDatabase[saved_recipe_count].time_unit_indices[i] =
										time_unit_indices[i];
							}

							saved_recipe_count++; // Increment the counter

							// --- Save to Flash [CRITICAL] ---
							SaveRecipesToFlash();

							// 2. SHOW SUCCESS FEEDBACK
							TFT_FillRect_Fast(10, 30, 300, 40, 0x07E0); // Green box
							TFT_DrawString(85, 42, "RECIPE SAVED!",
							ILI9341_COLOR_BLACK, 2);

							// 3. WAIT FOR USER TO SEE IT
							HAL_Delay(800);

							// 4. CRITICAL: WAIT FOR FINGER RELEASE
							// This stops the "OK" touch from being interpreted as a click on the next menu
							while (HAL_GPIO_ReadPin(GPIOC, TOUCH_IRQ_Pin)
									== GPIO_PIN_RESET)
								;

							// If ghosting
							//TFT_FillScreen(ILI9341_COLOR_GRAY);

							// 6. TRANSITION
							currentState = MENU_MAIN;
							Draw_Main_Menu();
							continue;
						}
					}
					// --- Regular Characters ---
					else {
						if (strlen(keyboard_buffer) < 15) {
							strcat(keyboard_buffer, keys[idx]);
						}
					}

					if (currentState == MENU_KEYBOARD) {
						TFT_FillRect_Fast(11, 36, 298, 38, 0x4208); // The Grey Box
						TFT_DrawString(20, 47, keyboard_buffer,
						ILI9341_COLOR_WHITE, 2);
						HAL_Delay(250);
					}
				}
			}
			//Load saved recipes back into synthesis menu
			else if (currentState == MENU_LOAD) {
				// Back Button
				if (ty >= 200 && tx >= 100 && tx <= 220) {
					currentState = MENU_MAIN;
					Draw_Main_Menu();
					HAL_Delay(300);
				}

				// Check for Recipe Slot Clicks
				if (saved_recipe_count > 0) {

					for (int i = 0; i < saved_recipe_count; i++) {
						int y_start = 50 + (i * 38); // Adjusted to match the Draw function
						int y_end = y_start + 32;

						// --- CHECK DELETE CROSS (X) HITBOX ---
						if (tx > 250 && tx < 310 && ty >= y_start
								&& ty <= y_end) {
							// Shift array to remove item i
							for (int j = i; j < saved_recipe_count - 1; j++) {
								recipeDatabase[j] = recipeDatabase[j + 1];
							}
							saved_recipe_count--;
							SaveRecipesToFlash();
							Draw_Load_Recipe_Menu();
							HAL_Delay(300);
							break;
						}

						// --- CHECK LOAD RECIPE HITBOX ---
						if (tx >= 10 && tx <= 240 && ty >= y_start
								&& ty <= y_end) {
							Load_Recipe_To_System(i);
							currentState = MENU_NEW_SYNTHESIS;
							Draw_New_Synthesis_Menu();
							HAL_Delay(300);
							break;
						}
					}
				}
			}
		}
	}
}

/* USER CODE BEGIN 4 */
void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim) {
	if (htim->Instance == TIM3) {

		// --- TRAY MOTOR LOGIC ---
		if (tray_is_moving) {
			uint8_t tray_stop_request = 0;

			if (tray_homing) {
				// PHASE 1: SENSOR SEARCH
				// 1. Ensure we have cleared the magnet area (Debounce)
				if (HAL_GPIO_ReadPin(TRAY_HALL_PORT, TRAY_HALL_PIN)
						== GPIO_PIN_SET) {
					tray_clear = 1;
				}
				// 2. Stop the moment we hit the magnet (Falling Edge)
				if (tray_clear
						&& HAL_GPIO_ReadPin(TRAY_HALL_PORT, TRAY_HALL_PIN)
								== GPIO_PIN_RESET) {
					tray_stop_request = 1;
				}

				// Safety Timeout: If 8000 steps pass without finding sensor, stop.
				if (tray_steps_remaining-- <= 0)
					tray_stop_request = 1;

			} else {
				// PHASE 2 & NORMAL MOVES: EXACT STEPS
				if (tray_steps_remaining-- <= 0) {
					tray_stop_request = 1;
				}
			}

			if (tray_stop_request) {
				tray_is_moving = 0;
				HAL_TIM_Base_Stop_IT(&htim3);
			} else {
				HAL_GPIO_WritePin(STEP_TRAY_PORT, STEP_TRAY_PIN, GPIO_PIN_SET);
				for (volatile int i = 0; i < 20; i++)
					; // Pulse width
				HAL_GPIO_WritePin(STEP_TRAY_PORT, STEP_TRAY_PIN,
						GPIO_PIN_RESET);
			}
		}

		// --- LEAD MOTOR LOGIC ---
		else if (lead_is_moving) {
			uint8_t lead_stop_request = 0;

			// Limit Switch Safety
			if (lead_dir == 1) { // Moving DOWN
				if (HAL_GPIO_ReadPin(LEAD_LIMIT_BTM_PORT,
				LEAD_LIMIT_BTM_PIN) == GPIO_PIN_RESET)
					lead_stop_request = 1;
				if (lead_mode == MOVE_SEEK_BUTT
						&& HAL_GPIO_ReadPin(SYRINGE_BUTT_PORT,
						SYRINGE_BUTT_PIN) == GPIO_PIN_RESET)
					lead_stop_request = 1;
			} else { // Moving UP
				if (HAL_GPIO_ReadPin(LEAD_LIMIT_TOP_PORT,
				LEAD_LIMIT_TOP_PIN) == GPIO_PIN_RESET)
					lead_stop_request = 1;
			}

			// Step Counter
			if (lead_mode == MOVE_EXACT_STEPS && lead_steps_remaining-- <= 0) {
				lead_stop_request = 1;
			}

			if (lead_stop_request) {
				lead_is_moving = 0;
